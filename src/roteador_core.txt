 ip access-group ACL_ADMIN in

interface GigabitEthernet0/0.20
 encapsulation dot1Q 20
 ip address 192.168.20.1 255.255.255.0
 ip access-group ACL_DEV in

interface GigabitEthernet0/0.30
 encapsulation dot1Q 30
 ip address 192.168.30.1 255.255.255.0

interface GigabitEthernet0/0.40
 encapsulation dot1Q 40
 ip address 192.168.40.1 255.255.255.0
 ip access-group ACL_IOT in

 encapsulation dot1Q 50

ip dhcp excluded-address 192.168.10.1 192.168.10.99
ip dhcp excluded-address 192.168.20.1 192.168.20.99
ip dhcp excluded-address 192.168.40.1 192.168.40.99
ip dhcp excluded-address 192.168.50.1 192.168.50.99

ip dhcp pool ADMIN
 network 192.168.10.0 255.255.255.0
 default-router 192.168.10.1
 dns-server 192.168.30.20

ip dhcp pool DEV
 network 192.168.20.0 255.255.255.0
 default-router 192.168.20.1
 dns-server 192.168.30.20

ip dhcp pool IOT
 network 192.168.40.0 255.255.255.0
 default-router 192.168.40.1
 dns-server 192.168.30.20

ip dhcp pool GUESTS
 network 192.168.50.0 255.255.255.0
 default-router 192.168.50.1
 dns-server 8.8.8.8

ip access-list extended ACL_ADMIN
 permit ip 192.168.10.0 0.0.0.255 any

ip access-list extended ACL_DEV
 permit tcp 192.168.20.0 0.0.0.255 192.168.30.0 0.0.0.255
 permit icmp 192.168.20.0 0.0.0.255 192.168.30.0 0.0.0.255
 deny ip 192.168.20.0 0.0.0.255 192.168.10.0 0.0.0.255
 permit ip 192.168.20.0 0.0.0.255 any

ip access-list extended ACL_IOT
 permit tcp 192.168.40.0 0.0.0.255 host 192.168.30.30 eq 1883
 permit tcp 192.168.40.0 0.0.0.255 host 192.168.30.30 eq 8883
 permit udp 192.168.40.0 0.0.0.255 host 192.168.30.20 eq 53
 permit udp 192.168.40.0 0.0.0.255 host 192.168.30.20 eq 123
 deny ip 192.168.40.0 0.0.0.255 192.168.0.0 0.0.255.255
 permit ip 192.168.40.0 0.0.0.255 any

ip access-list extended ACL_GUESTS
 deny ip 192.168.50.0 0.0.0.255 192.168.0.0 0.0.255.255
 permit udp 192.168.50.0 0.0.0.255 any eq 53
 permit tcp 192.168.50.0 0.0.0.255 any eq 80
 permit tcp 192.168.50.0 0.0.0.255 any eq 443
 permit ip 192.168.50.0 0.0.0.255 any

end
